# Python dependencies for the e-paper display client (display/epaper_client.py).
# Install on the Raspberry Pi 5:  pip3 install -r display/requirements.txt
#
# Networking + JSON use only the Python standard library (urllib, json) — no
# `requests` needed. The deps below are the rendering and panel-driver stack.

# Image rendering for the 1-bit framebuffer.
Pillow>=10.0.0

# Waveshare e-Paper driver. The PyPI package is `waveshare-epd`; it pulls in the
# Pi 5-compatible GPIO backend (gpiozero + lgpio). If pip can't find it for your
# OS, vendor the official driver instead (see README "Driver" section).
waveshare-epd>=1.0.0 ; platform_machine == "aarch64" or platform_machine == "armv7l"

# GPIO + SPI backends the Waveshare driver uses on the Raspberry Pi 5.
# (RPi.GPIO does NOT work on the Pi 5 — gpiozero/lgpio is required.)
gpiozero>=2.0 ; platform_machine == "aarch64" or platform_machine == "armv7l"
lgpio>=0.2 ; platform_machine == "aarch64" or platform_machine == "armv7l"
spidev>=3.5 ; platform_machine == "aarch64" or platform_machine == "armv7l"
