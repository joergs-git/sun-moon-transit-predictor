# Python dependencies for the e-paper display client (display/epaper_client.py).
#
# On a Raspberry Pi 5 the easiest path is the installer, which also fetches the
# Waveshare driver from GitHub and sets up SPI + groups:
#     bash scripts/install-pi5.sh --with-display
#
# To install the pip-installable parts by hand (Debian/bookworm needs
# --break-system-packages):
#     pip3 install --break-system-packages -r display/requirements.txt
#
# Networking + JSON use only the Python standard library (urllib, json) — no
# `requests` needed.

# Image rendering for the 1-bit framebuffer.
Pillow>=10.0.0

# Pi 5 GPIO/SPI backends. RPi.GPIO does NOT work on the Pi 5 — gpiozero with the
# lgpio factory is required; rpi-lgpio provides the RPi.GPIO API on top of lgpio
# for driver code that still imports it.
gpiozero>=2.0 ; platform_machine == "aarch64" or platform_machine == "armv7l"
lgpio>=0.2 ; platform_machine == "aarch64" or platform_machine == "armv7l"
rpi-lgpio>=0.6 ; platform_machine == "aarch64" or platform_machine == "armv7l"
spidev>=3.5 ; platform_machine == "aarch64" or platform_machine == "armv7l"

# NOTE: the Waveshare panel driver (`waveshare_epd`) is NOT on PyPI. Install it
# from the official repo (the installer's --with-display does this for you):
#     git clone --depth 1 https://github.com/waveshareteam/e-Paper /tmp/e-Paper
#     pip3 install --break-system-packages --no-deps /tmp/e-Paper/RaspberryPi_JetsonNano/python
